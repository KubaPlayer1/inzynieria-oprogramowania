using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms;

namespace Saper
{
    internal class PanelGame : Panel
    {
        ButtonField[,] board;
        private const int size = 30;


        public PanelGame(int width, int height, int bombCount)
        {
            this.AutoSize = true;
            board = new ButtonField[width, height];
            for (int x = 0; x < width; x++)
            {
                for (int y = 0; y < height; y++)
                {
                    board[x, y] = new ButtonField();
                    board[x, y].Size = new Size(size, size);
                    board[x, y].Location = new Point(size * x, size * y);

                    this.Controls.Add(board[x, y]);
                }
            }
        }
    }
}


using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Saper
{
    internal class ButtonField : Button
    {
        private bool isBomb;
        private int bombCounter;
        private bool isOpen;

        public ButtonField() { }
        public ButtonField(bool isBomb, int bombCounter, bool isOpen)
        {
            this.isBomb = isBomb;
            this.bombCounter = bombCounter;
            this.isOpen = isOpen;
        }
    }
}
